<script>
  const filters = document.querySelectorAll('.pub-filter');
  filters.forEach(btn => {
    btn.addEventListener('click', () => {
      filters.forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      const type = btn.dataset.filter;
      document.querySelectorAll('.pub-item').forEach(item => {
        item.classList.toggle('hidden', type !== 'all' && item.dataset.type !== type);
      });
      document.querySelectorAll('.pub-group-label').forEach(label => {
        const group = label.dataset.group;
        const visible = [...document.querySelectorAll(`.pub-item[data-type="${group}"]`)].some(i => !i.classList.contains('hidden'));
        label.style.display = (type === 'all' || visible) ? '' : 'none';
      });
    });
  });
</script>
